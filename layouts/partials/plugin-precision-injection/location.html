{{- /* This partial expects a dictionary as its parameter. keys: 'location': The key into the Locations map for the list of files to inject. Pass the empty string to inject all files not assigned a location. 'context': The page context (aka "the dot"). */}} {{- /* Retrieve the configuration and pull out what we need. */ -}} {{- $Config := .context.Scratch.Get "plugin-precision-injection" -}} {{- $Locations := $Config.Locations -}} {{- $Assigned := $Config.Assigned -}} {{- $HTMLComments := $Config.HTMLComments -}} {{- $Filters := $Config.Filters -}} {{- /* Retrieve the full list of plugin html includes. */ -}} {{- $Files := site.Params.plugins_html -}} {{- /* Grab the path for the page being generated */ -}} {{- $Path := (urls.Parse .context.Permalink).Path -}} {{- /* When a location key has been provided, limit the list of files to those that appear in the file list set for that key. */ -}} {{- if .location -}} {{- $Files = $Files | intersect (index $Locations .location) -}} {{- /* When the location key is missing or empty, limit the list of files to those that do no appear anywhere in the $Locations map. */ -}} {{- else -}} {{- $Files = $Files | complement $Assigned -}} {{- end -}} {{- /* with .location */ -}} {{- /* Create a list containing the items in $Files whose existence has been confirmed. */ -}} {{- $ConfirmedFiles := slice -}} {{- range $file := $Files -}} {{- if (templates.Exists (path.Join "partials" $file)) -}} {{- $ConfirmedFiles = $ConfirmedFiles | append $file -}} {{- end -}} {{- end -}} {{- /* Create a list containing the items in $ConfirmedFiles that are not filtered out via the $Filters map */ -}} {{- $FilteredFiles := slice -}} {{- range $file := $ConfirmedFiles -}} {{- /* Retrieve the $Filters entry for $file (if it exists). */ -}} {{- $whitelist := index $Filters $file -}} {{- /* To simplify logic, ensure $whitelist is a slice. */ -}} {{- if (and $whitelist (not (reflect.IsSlice $whitelist))) -}} {{- $whitelist = slice $whitelist -}} {{- end -}} {{- /* Append $file if it has no entry or $Path is in the list. */ -}} {{- if (or (not $whitelist) (in $whitelist $file)) -}} {{- $FilteredFiles = $FilteredFiles | append $file -}} {{- end -}} {{- end -}} {{- /* range $file := $ConfirmedFiles */ -}} {{- /* Inject the filtered list of files. */ -}} {{- range $file := $FilteredFiles -}} {{- /* Insert HTML comment with $file if so flagged */ -}} {{- if $HTMLComments -}} {{- printf "\n\n" $file | safeHTML -}} {{- end -}} {{- /* Inject the file. */ -}} {{ partial $file $.context }} {{- end -}} {{- /* range $file := $FilteredFiles */ -}}
